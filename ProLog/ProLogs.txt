1- Who is a nephew

	isNephew(X) :- parent(Z, X), male(X), isSibling(Z,_).

2- Who is a son

	isSon(X) :- parent(_, X), male(X).

3- Who is a daughter

	isDaughter(X) :- parent(_, X), female(X).

4- Who is a parent

	/*Based on InClass Slides*/
	isParent(X,Y) :- parent(X,Z),parent(Y,Z), X\==Y. 

5- Who is a grandparent

	/*Based on InClass Slides*/
	isGrandparent(X, Y) :- parent(X, Z), parent(Z, Y).

6- Who is a great grand parent

	isGreatGrandparent(X,Y) :- parent(X,Z), parent(Z,A), parent(A, Y).

7- Who is either an Aunt or an uncle

	/*Based on InClass Slides with personal alternations*/
	auncle(X):- isSibling(X,Z), isParent(Z, A), A\==Z.

8- Who are siblings

	/*Based on InClass Slides*/
	isSibling(X,Y) :- parent(Z, X), parent(Z,Y), X\==Y.
